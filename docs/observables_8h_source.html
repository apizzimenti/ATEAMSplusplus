79 inline std::vector<int> satisfied(
80 Complex<T>* complex,
81 SparseVector<T> f,
82 Field F,
83 int dimension
84 ) {
85 // Perform the matrix multiplication.
86 SparseVector<T> coefficients = SparseRREF::sparse_mat_dot_sparse_vec<T,index_t>(
87 complex->Coboundary.Matrices[dimension], f, F
88 );
89 coefficients.compress();
90
91 // Create a vector containing the indices of nonzero entries.
92 std::vector<int> unsatisfied(coefficients.size());
93 for (int i=0; i < unsatisfied.size(); i++) unsatisfied[i] = (int)coefficients(i);
94
95 // Create a vector containing the indices of *all* entries, then take
96 // the difference.
97 std::vector<int> totality(complex->Cells[dimension]);
98 std::iota(totality.begin(), totality.end(), 0);
99
100 std::vector<int> satisfied;
101 std::set_difference(
102 totality.begin(), totality.end(),
103 unsatisfied.begin(), unsatisfied.end(),
104 inserter(satisfied, satisfied.begin())
105 );
106
107 return satisfied;
108 }
