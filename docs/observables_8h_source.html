84 template <typename RingLike>
85 inline std::vector<int> satisfied(
86 complexes::Complex<RingLike>* complex,
87 SparseVector<RingLike> f,
88 Ring* R,
89 int dimension
90 ) {
91 // Perform the matrix multiplication.
92 SparseVector<RingLike> coefficients = SparseRREF::sparse_mat_dot_sparse_vec<typename RingLike::dtype,INDEX>(
93 complex->Coboundary.Matrices[dimension], f, R->ring
94 );
95 coefficients.compress();
96
97 // Create a vector containing the indices of nonzero entries.
98 std::vector<int> unsatisfied(coefficients.size());
99 for (int i=0; i < unsatisfied.size(); i++) unsatisfied[i] = (int)coefficients(i);
100
101 // Create a vector containing the indices of *all* entries, then take
102 // the difference.
103 std::vector<int> totality(complex->Cells[dimension]);
104 std::iota(totality.begin(), totality.end(), 0);
105
106 std::vector<int> satisfied;
107 std::set_difference(
108 totality.begin(), totality.end(),
109 unsatisfied.begin(), unsatisfied.end(),
110 inserter(satisfied, satisfied.begin())
111 );
112
113 return satisfied;
114 }
