23 template <typename RingLike>
24 Glauber<RingLike>::State Glauber<RingLike>::sample(
25 int t,
26 Glauber<RingLike>::State& state,
27 arithmetic::ThreadOptions& options
28 ) {
29 // Compute the temperature and probability of including particular (d-1)-cells.
30 double temp = this->temperatureFunction(t);
31
32 // Sample a (d-1)-cell at uniform random and change its spin.
33 INDEX flippedIndex = this->indexuniform(this->RNG);
34 SparseVector<RingLike> flipped(state.cochain);
35
36 // Overwrite the entry? Not sure if this actually overwrites it or not.
37 flipped.push_back(
38 (INDEX)flippedIndex,
39 (typename RingLike::dtype)this->intuniform(this->RNG)
40 );
41
42 // Compute the energies of the cochains.
43 // TODO this is gonna be really slow, so we should work out a better way to
44 // compute these.
45 int original = statistics::energy(this->complex, state.cochain, this->coefficients, this->dimension);
46 int updated = statistics::energy(this->complex, flipped, this->coefficients, this->dimension);
47
48 // Comptue the probability of accepting the new cochain; perform the Metropolis
49 // step.
50 double random = this->unituniform(this->RNG);
51 double diff = min(1.0, exp(temp*(original-updated)));
52
53 if (random < diff) {
54 state.cochain = flipped;
55 state.energy = updated;
56 } else {
57 state.energy = original;
58 }
59
60 state.t = t;
61
62 return state;
63 }

66 template <typename RingLike>
67 Glauber<RingLike>::State Glauber<RingLike>::initialize(Glauber<RingLike>::State& state) {
68 size_t dimension = this->dimension-1;
69 int N = this->complex->Cells[dimension];
70
71 SparseVector<RingLike> cochain;
72 for (int i=0; i < N; i++) cochain.push_back(
73 (INDEX)i, (typename RingLike::dtype)this->intuniform(this->RNG)
74 );
75
76 cochain.compress();
77 state.cochain = cochain;
78
79 return state;
80 }

92 template <typename RingLike>
93 Glauber<RingLike>::Glauber(
94 complexes::Complex<RingLike>* complex,
95 Ring* R,
96 int dimension,
97 std::function<double(int)> temperatureFunction,
98 bool DEBUG
99 ) : Model<RingLike,SparseVector>(R, dimension, DEBUG) {
100 this->temperatureFunction = temperatureFunction;
101 this->complex = complex;
102
103 // Determine the field and build the boundary matrices for the Complex.
104 this->complex->constructBoundaryMatrices(this->coefficients);
105
106 // Initialize a random number generator.
107 std::random_device rd;
108 this->RNG = std::mt19937(rd());
109 this->unituniform = std::uniform_real_distribution<double>(0,1);
110 this->indexuniform = std::uniform_int_distribution<int>(0, this->complex->Cells[this->dimension-1]);
111
112 this->intuniform = std::uniform_int_distribution<int>(
113 0,
114 this->coefficients->characteristic > 0 ? this->coefficients->characteristic : 2
115 );
116 };
