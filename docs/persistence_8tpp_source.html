24 template <typename RingLike>
25 SparseMatrix<RingLike> reindexSparseBoundaryMatrix(
26 complexes::Complex<RingLike>* complex,
27 vector<int> filtration,
28 int dimension
29 ) {
30 // Construct an index mapping.
31 vector<int> remapping(filtration.size(), 0);
32 for (int t=0; t < filtration.size(); t++) remapping[filtration[t]] = t;
33
34 // Wow, this is going to be annoying.
35 SparseMatrix<RingLike> Full = complex->Coboundary.Full;
36 SparseMatrix<RingLike> Reindexed(Full.nrow, Full.ncol);
37
38 int startDimension = complex->Breaks[dimension][0];
39 int stopDimension = complex->Breaks[dimension][1];
40
41 for (int t=0; t < Full.nrow; t++) {
42 if ((startDimension <= t) && (t < stopDimension)) {
43 Reindexed.rows[t] = Full.rows[filtration[t]];
44 } else {
45 SparseVector<RingLike>& row = Reindexed.rows[t];
46 SparseVector<RingLike> orow = Full.rows[t];
47
48 for (int i=0; i < orow.size(); i++) {
49 row.push_back(
50 (INDEX)remapping[orow(i)],
51 (typename RingLike::dtype)orow[i]
52 );
53 }
54 }
55 }
56
57 Reindexed.compress();
58 return Reindexed;
59 }

63 template <typename RingLike>
64 vector<int> PHATPersistence(
65 complexes::Complex<RingLike>* complex,
66 vector<int> filtration,
67 int dimension
68 ) {
69 // The filtration specifies the order in which we add the cells of all
70 // dimensions. Create a map that specifies to which position each cell was
71 // moved. For example, if the filtration has
72 //
73 // [... 12, 9, 10, 19, ...]
74 //
75 // that was originally
76 // [... 9, 10, 11, 12, ...]
77 //
78 // then the mapping should have entries
79 //
80 // {... 12: 9, 9: 10, 10: 11, 19: 12, ...}
81
82 // TODO optimize, this takes way too long
83 vector<int> remapping(filtration.size(), 0);
84 for (int t=0; t < filtration.size(); t++) remapping[filtration[t]] = t;
85
86 PHATBoundaryMatrix boundary;
87 PHATColumn column;
88 int faces;
89
90 int startDimension = complex->Breaks[dimension][0];
91 int stopDimension = complex->Breaks[dimension][1];
92
93 boundary.set_num_cols(complex->Boundary.Flat.size());
94
95 // this also takes an insane amount of time; no need to go all the way
96 // up.
97 for (int t=0; t < complex->Boundary.Flat.size(); t++) {
98 faces = complex->Boundary.Flat[t].size();
99
100 // Check whether we're in the dimension of the cells on which we're
101 // percolating; if so, add the cells in the order specified in the
102 // filtration. If we're in the percolation dimension + 1, re-index the
103 // (d+1)-dim cells to reflect the ordering of the d-dimensional cells.
104 // Otherwise, do nothing.
105 if ((startDimension <= t) && (t < stopDimension)) {
106 for (int j=0; j < faces; j++) column.push_back(complex->Boundary.Flat[filtration[t]][j]);
107 } else {
108 for (int j=0; j < faces; j++) column.push_back(remapping[complex->Boundary.Flat[t][j]]);
109 }
110
111 std::sort(column.begin(), column.end());
112
113 boundary.set_dim(t, faces/2);
114 boundary.set_col(t, column);
115 column.clear();
116 }
117
118 // Compute the persistence pairs and populate a Vector to return to the
119 // user. We know when the giant(/essential) cycles are born because they
120 // do not appear in the list of births or deaths.
121 PHATPairs pairs;
122 set<int> all;
123 phat::compute_persistence_pairs<PHATTwist>(pairs, boundary);
124 pairs.sort();
125
126 for (phat::index i=0; i < pairs.get_num_pairs(); i++) {
127 all.insert(pairs.get_pair(i).first);
128 all.insert(pairs.get_pair(i).second);
129 }
130
131 int low = complex->Breaks[dimension][0], high = complex->Breaks[dimension][1];
132 vector<int> essential;
133
134 for (int t=0; t < filtration.size(); t++) {
135 if (!all.contains(filtration[t]) && (low <= filtration[t] && filtration[t] < high)){
136 essential.push_back(filtration[t]);
137 }
138 }
139
140 return essential;
141 }

144 template <typename RingLike>
145 vector<int> twistPersistence(
146 complexes::Complex<RingLike>* complex,
147 vector<int> filtration,
148 Ring* R,
149 int dimension
150 ) {
151 // Doing row operations on the coboundary is equivalent to column operations
152 // on the boundary.
153 SparseMatrix<RingLike> Full = reindexSparseBoundaryMatrix<RingLike>(complex, filtration, dimension);
154 SparseVector<RingLike> youngestFace;
155
156 // Track which column is to be added next; track which ones are marked.
157 vector<int> nextColumnAdded(complex->size(), 0);
158 set<int> marked;
159
160 // Top dimension of the complex; indices at which we stop and start.
161 int topDimension = complex->Cells.size(), lowestCellIndex, highestCellIndex;
162 typename RingLike::dtype youngestFaceCoefficient;
163
164 for (int d=topDimension-1; d > dimension-1; d--) {
165 lowestCellIndex = complex->Breaks[d][0];
166 highestCellIndex = (d+1 >= complex->Cells.size()) ? complex->size() : complex->Breaks[d][1];
167
168 for (int j=lowestCellIndex; j < highestCellIndex; j++) {
169 SparseVector<RingLike>& cell = Full.rows[j];
170
171 while (cell.size() > 0 && nextColumnAdded[youngestFaceIndexOf<RingLike>(cell)] != 0) {
172 youngestFace = Full.rows[nextColumnAdded[youngestFaceIndexOf<RingLike>(cell)]];
173 youngestFaceCoefficient = *youngestFace.find(youngestFaceIndexOf<RingLike>(cell));
174
175 // TODO parallelization stuff here? Can't go across columns, so maybe
176 // within the column? SparseRREF/FLINT probably do that already tho.
177 typename RingLike::dtype s = scalar_neg(scalar_inv(youngestFaceCoefficient, R->ring), R->ring);
178
179 arithmetic::SparseVectorRescaling<RingLike>(s, youngestFace, R);
180 arithmetic::SparseVectorAddition(cell, youngestFace, R);
181
182 cell.compress();
183 }
184
185 if (cell.size() > 0) {
186 nextColumnAdded[youngestFaceIndexOf<RingLike>(cell)] = j;
187 Full.rows[youngestFaceIndexOf<RingLike>(cell)].zero();
188 } else {
189 marked.insert(j);
190 }
191 }
192 }
193
194 int low = complex->Breaks[dimension][0], high = complex->Breaks[dimension][1];
195 vector<int> essential;
196
197 for (auto k : marked) {
198 if (nextColumnAdded[k] == 0 && (low <= k && k < high)) essential.push_back(k);
199 }
200
201 return essential;
202 };

205 template <typename RingLike>
206 std::vector<int> persistence(
207 complexes::Complex<RingLike>* complex,
208 std::vector<int> filtration,
209 Ring* R,
210 int dimension
211 ) {
212 vector<int> essential;
213 if (R->characteristic < 3) essential = PHATPersistence<RingLike>(complex, filtration, dimension);
214 else essential = twistPersistence<RingLike>(complex, filtration, R, dimension);
215
216 std::sort(essential.begin(), essential.end());
217 return essential;
218 };
