Detailed Description

template<typename RingLike, template< typename > typename VectorLike>
class ATEAMS::statistics::Chain< RingLike, VectorLike >

Simulates the Markov chains specified by each Model.

Template Parameters
RingLikeA coefficient Ring type, like Zp or Q.
VectorLikeStorage unit compatible with RingLike, like SparseVector or DenseVector.

Example Usage

using namespace ATEAMS;
using Model = models::Bernoulli;
using Parameters = models::ModelParameters;
using Complex = complexes::Cubical<Model::RingType>;
using State = models::ModelState<Model::RingType,Model::VectorType>;
using Chain = statistics::Chain<Model::RingType,Model::VectorType>;
Complex plex({4,4,4,4});
Parameters params;
params.p = 0.5;
params.dimension = 2;
Model percolation(&plex, parameters);
Chain M(&percolation, 1000);
for (State state : M.simulate()) {

}
