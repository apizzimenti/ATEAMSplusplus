18 template <typename RingLike>
19 SparseVector<RingLike> randomLinearCombination(
20 SparseMatrix<RingLike> K,
21 Ring* R,
22 std::uniform_int_distribution<int>& intuniform,
23 std::mt19937& RNG
24 ) {
25 // Create a vector of random coefficients, multiply the kernel on the right, done.
26 SparseVector<RingLike> rcoeffs;
27 for (int i=0; i < K.ncol; i++) rcoeffs.push_back((INDEX)i, (typename RingLike::dtype)intuniform(RNG));
28 rcoeffs.compress();
29
30 return SparseRightMultiplication<RingLike>(K, rcoeffs, R);
31 }

34 template <typename RingLike>
35 SparseVector<RingLike> submatrixKernelSample(
36 SparseMatrix<RingLike> coboundary,
37 Ring* R,
38 set<size_t> exclude,
39 std::uniform_int_distribution<int>& intuniform,
40 std::mt19937& RNG,
41 ThreadOptions& options,
42 bool DEBUG
43 ) {
44 // Shrink the coboundary matrix to the appropriate size.
45 for (auto i : exclude) coboundary[i].zero();
46 coboundary.clear_zero_row(); // TODO check the performance of this!
47 coboundary.compress();
48
49 SparseMatrix<RingLike> kernel;
50
51 if (DEBUG) {
52 // RREF (i.e. find the pivots for) the subcoboundary matrix.
53 cerr << std::format("RREFing {}x{} coboundary matrix", coboundary.nrow, coboundary.ncol) << endl;
54 SparsePivots pivots = SparseMatrixRREF<RingLike>(coboundary, R, options);
55
56 // Find a kernel for the subcoboundary matrix.
57 cerr << std::format("computing kernel of {}x{} coboundary matrix", coboundary.nrow, coboundary.ncol) << endl;
58 kernel = SparseMatrixRREFKernel<RingLike>(coboundary, R, pivots, options);
59 } else {
60 kernel = SparseMatrixKernel<RingLike>(coboundary, R, options);
61 }
62
63 // Return a random linear combination of the columns.
64 return randomLinearCombination<RingLike>(kernel, R, intuniform, RNG);
65 }
