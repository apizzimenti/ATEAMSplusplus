16 Bernoulli::State Bernoulli::sample(
17 int t,
18 Bernoulli::State& state,
19 arithmetic::ThreadOptions& options
20 ) {
21 // If we're debugging, check that the inclusion probability is a probability
22 // to begin with.
23 double p = this->p;
24 int d = this->dimension;
25
26 if (this->DEBUG) {
27 cerr << std::format("verifying 0 ≤ {} ≤ 1", p) << endl;
28 assert((0 <= p) && (p <= 1));
29 }
30
31 // Then decide which cells we're including/excluding.
32 vector<int> include;
33 vector<int> exclude;
34
35 for (int i=0; i < this->complex->Cells[d]; i++) {
36 if (this->unituniform(this->RNG) < p) {
37 include.push_back(i);
38 } else {
39 exclude.push_back(i);
40 }
41 }
42
43 int included = include.size();
44 state.includes = include;
45
46 // Construct the filtration.
47 int stop = this->complex->Breaks[d][0];
48 int start = this->complex->Breaks[d][1];
49 int offset = (d > 0) ? this->complex->Offsets[d-1] : 0;
50
51 // Fill in all the indices for cells of dimension not equal to d.
52 for (int j=0; j < stop; j++) this->filtration[j] = j;
53 for (int j=start; j < this->complex->Offsets[d+1]; j++) this->filtration[j] = j;
54
55 // Shuffle the included indices.
56 std::shuffle(include.begin(), include.end(), this->RNG);
57
58 for (int j=0; j < include.size(); j++) this->filtration[j+offset] = include[j]+offset;
59 for (int j=0; j < exclude.size(); j++) this->filtration[j+offset+include.size()] = exclude[j]+offset;
60
61 // Now, compute the persistence times, then filter over them to capture only
62 // the ones within the right window.
63 vector<int> essential = topology::persistence<Z2>(this->complex, this->filtration, this->coefficients, d);
64 std::erase_if(essential, [stop, included](int t) { return !((stop <= t) && (t < stop+included)); });
65
66 if (this->DEBUG) {
67 std::cerr << "outputting filtration to stdout..." << std::endl;
68 printvector<int>(this->filtration);
69
70 std::cerr << "outputting essential cycles to stdout..." << std::endl;
71 printvector<int>(essential);
72 }
73
74 // Knock off the offset value, set state, and done.
75 std::for_each(essential.begin(), essential.end(), [offset](int &k) { k -= offset; });
76 state.essential = essential;
77 state.rank = essential.size();
78 state.t = t;
79
80 return state;
81 }
